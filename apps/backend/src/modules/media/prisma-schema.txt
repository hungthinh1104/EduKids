model Media {
  id                   String           @id @default(cuid())
  mediaType            String           // IMAGE, AUDIO, VIDEO
  context              String           // VOCABULARY, TOPIC, QUIZ, PROFILE, GENERAL
  originalFilename     String
  fileSize             Int
  mimeType             String
  status               String           // PENDING, PROCESSING, COMPLETED, FAILED
  rawUrl               String?          // Temporary storage path
  optimizedUrl         String?          // Cloudinary URL after processing
  thumbnailUrl         String?          // Thumbnail for images/videos
  cloudinaryPublicId   String?          @unique
  description          String?          @db.Text
  altText              String?
  uploadedBy           String
  uploadedAt           DateTime         @default(now())
  processedAt          DateTime?
  errorMessage         String?          @db.Text
  metadata             Json?            // dimensions, duration, format, etc.
  
  // Relations
  admin                User             @relation(fields: [uploadedBy], references: [id])
  
  // Indexes
  @@index([mediaType])
  @@index([context])
  @@index([status])
  @@index([uploadedBy])
  @@index([uploadedAt])
  @@map("media")
}
